pipeline {
    agent any

    stages {
        // Git Clone 
        stage('Git Clone') {
            steps {
                git branch:'main', url: 'https://github.com/beening01/jenkins.git'
            }
                post {
			      success {
			                    echo "Success Cloned Repository"
			                }
			
			     failure {
			                    error "Fail Cloned Repository"
			                }
			        }
        }
        
    }
}
